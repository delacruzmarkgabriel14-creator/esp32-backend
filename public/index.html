const express = require("express"); const app = express(); app.use(express.json()); // Serve frontend app.use(express.static("public")); // Stores latest ESP32 data let latestData = { temperature: null, humidity: null, fan: null }; // API routes app.post("/api/data", (req, res) => { latestData = req.body; console.log("Received:", latestData); res.send("Data stored"); }); app.get("/api/data", (req, res) => { res.json(latestData); }); const PORT = process.env.PORT || 10000; app.listen(PORT, () => console.log("Server running on port", PORT));
