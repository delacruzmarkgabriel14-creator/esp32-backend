const express = require("express"); const app = express(); const port = process.env.PORT || 3000; // For parsing JSON app.use(express.json()); // Serve static files from "public" folder app.use(express.static("public")); // Example data storage let latestData = { temperature: null, humidity: null, fan: null }; // API route for ESP32 to POST data app.post("/api/data", (req, res) => { latestData = req.body; console.log("Received:", latestData); res.send("Data stored"); }); // API route for frontend to GET latest data app.get("/api/data", (req, res) => { res.json(latestData); }); // Start server app.listen(port, () => { console.log(`Server running on port ${port}`); });
